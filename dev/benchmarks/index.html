julia> using Logging

julia> Logging.LogLevel(Error)

function subscribe_for_one_second()
    tm = Timer(1)
    counter = 0
    start = nothing
    sub = subscribe("foo") do
        if isnothing(start)
            start = time()
        end
        counter += 1
    end
    # while counter < 20000000
    #     sleep(1)
    # end
    wait(tm)
    unsubscribe(sub)
    if counter == 0
        @info "No messages"
    else
        @info "Processed $counter messages in $(time() - start) s."
    end
end
